cmake_minimum_required(VERSION 2.6)
find_library(HAVE_LIBJANSSON NAMES jansson)
if(HAVE_LIBJANSSON)
	set(ExtraLibs ${ExtraLibs} ${HAVE_LIBJANSSON})
	set(PACKAGE_EXTRA_LIBRARY ${ExtraLibs})
	set(PACKAGE_SOURCE_CODE jansson_glue.c)
	set(PACKAGE_SCRIPT_CODE jansson_glue.k)
	add_konoha_package(jansson)
endif(HAVE_LIBJANSSON)
