cmake_minimum_required(VERSION 2.6)

if(HAVE_LIBJSON)
	set(ExtraLibs ${ExtraLibs} ${HAVE_LIBJSON})
	set(PACKAGE_EXTRA_LIBRARY ${ExtraLibs})
	set(PACKAGE_SOURCE_CODE json_glue.c)
	set(PACKAGE_SCRIPT_CODE json_glue.k)
	add_konoha_package(konoha.json)
endif(HAVE_LIBJSON)
